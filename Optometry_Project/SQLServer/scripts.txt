
CREATE TABLE patients (
    patient_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    birth_date DATE,
    phone VARCHAR(20),
    email VARCHAR(50)
);


CREATE TABLE optometrists (
    optometrist_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    license_number VARCHAR(30) UNIQUE NOT NULL,
    phone VARCHAR(20),
    email VARCHAR(50)
);

CREATE TABLE appointments (
    appointment_id SERIAL PRIMARY KEY,
    patient_id INT NOT NULL REFERENCES patients(patient_id),
    optometrist_id INT NOT NULL REFERENCES optometrists(optometrist_id),
    appointment_date TIMESTAMP NOT NULL,
    notes TEXT
);


CREATE TABLE eye_exams (
    exam_id SERIAL PRIMARY KEY,
    appointment_id INT NOT NULL REFERENCES appointments(appointment_id),
    exam_date TIMESTAMP NOT NULL,
    vision_left VARCHAR(20),
    vision_right VARCHAR(20),
    diagnosis TEXT
);


CREATE TABLE prescriptions (
    prescription_id SERIAL PRIMARY KEY,
    exam_id INT NOT NULL REFERENCES eye_exams(exam_id),
    prescription_date TIMESTAMP NOT NULL,
    details TEXT
);


CREATE TABLE products (
    product_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    stock INT DEFAULT 0
);


CREATE TABLE sales (
    sale_id SERIAL PRIMARY KEY,
    product_id INT NOT NULL REFERENCES products(product_id),
    quantity INT NOT NULL,
    total DECIMAL(10,2) NOT NULL,
    sale_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE sales_time (
    sale_time_id SERIAL PRIMARY KEY,
    sale_id INT NOT NULL REFERENCES sales(sale_id),
    start_time TIMESTAMP,
    end_time TIMESTAMP
);
